% Main Section Command
\newcommand{\mysection}[1]{%
    \addcontentsline{toc}{section}{#1} % Adds to Table of Contents
    \vspace{1em} 
    {\large \textbf{\strut #1}}\\[-0.8em]
    \rule{\linewidth}{0.8pt}\par\vspace{0.5em}
}

% Subsection Command
\newcommand{\mysubsection}[1]{%
    \addcontentsline{toc}{subsection}{#1} % Adds to Table of Contents
    \vspace{0.8em}
    {\normalsize \textbf{\strut #1}}\\[-0.9em]
    \rule{\linewidth}{0.4pt}\par\vspace{0.4em}
}

}
